class Solution {
    public int peakIndexInMountainArray(int[] nums) {
        int max = nums[0];
        int flag = 0;

        for (int i = 1; i < nums.length; i++) {
            if (nums[i] > max) {
                max = nums[i];
                flag = i;
            }
        }
        return flag;
    }
}
